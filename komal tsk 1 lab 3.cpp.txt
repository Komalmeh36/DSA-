#include <iostream>
#include <string>
using namespace std;

struct Product {
    string name;
    int code;
    float price;
};

int main() {
    int n;
    cout << "Enter number of products: ";
    cin >> n;

    Product* products = new Product[n];  // dynamic array

    for (int i = 0; i < n; i++) {
        cout << "\nEnter details for product " << i + 1 << endl;
        cout << "Name: ";
        cin >> products[i].name;
        cout << "Code: ";
        cin >> products[i].code;
        cout << "Price: ";
        cin >> products[i].price;
    }

    cout << "\nProduct List:\n";
    for (int i = 0; i < n; i++) {
        cout << "Name: " << products[i].name
             << ", Code: " << products[i].code
             << ", Price: " << products[i].price << endl;
    }

    delete[] products;   // free memory
    products = nullptr;

    return 0;
}
