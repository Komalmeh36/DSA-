#include <iostream>
using namespace std;

int main() {
    // Dynamic allocation
    char* ch = new char;

    // Store value
    cout << "Enter a character: ";
    cin >> *ch;

    // Print value
    cout << "Stored character: " << *ch << endl;

    // Deallocate memory
    delete ch;
    ch = nullptr;

    return 0;
}
