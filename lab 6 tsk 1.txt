#include <iostream>
using namespace std;

class BookNode {
public:
    string title;
    float price;
    int edition;
    int pages;
    BookNode* next;

    BookNode(string t, float p, int e, int pg) {
        title = t;
        price = p;
        edition = e;
        pages = pg;
        next = NULL;
    }
};

class Stack {
    BookNode* top;

public:
    Stack() {
        top = NULL;
    }

    void push(string title, float price, int edition, int pages) {
        BookNode* newNode = new BookNode(title, price, edition, pages);
        newNode->next = top;
        top = newNode;

        cout << "Book pushed successfully\n";
    }

    void pop() {
        if (top == NULL) {
            cout << "Stack Underflow\n";
            return;
        }

        BookNode* temp = top;
        top = top->next;

        cout << "Removed Book: " << temp->title << endl;

        delete temp;
    }

    void peek() {
        if (top == NULL) {
            cout << "Stack is Empty\n";
            return;
        }

        cout << "\nTop Book Details:\n";
        cout << "Title: " << top->title << endl;
        cout << "Price: " << top->price << endl;
        cout << "Edition: " << top->edition << endl;
        cout << "Pages: " << top->pages << endl;
    }

    void display() {
        BookNode* temp = top;

        while (temp != NULL) {
            cout << temp->title << " -> ";
            temp = temp->next;
        }
        cout << "NULL\n";
    }
};

int main() {

    Stack s;

    s.push("DSA", 500, 3, 450);
    s.push("Operating System", 650, 2, 600);
    s.push("Database", 550, 4, 500);
    s.push("Computer Networks", 700, 1, 480);
    s.push("AI", 800, 2, 550);

    s.peek();

    cout << "\nStack Elements:\n";
    s.display();

    cout << "\nPop Operation\n";
    s.pop();

    cout << "\nStack After Pop:\n";
    s.display();

}
