#include <iostream>
using namespace std;

int main() {
    int n, count = 0;

    cout << "Enter size of array: ";
    cin >> n;

    int arr[n];

    cout << "Enter elements:\n";
    for(int i = 0; i < n; i++) {
        cin >> arr[i];

        if(arr[i] % 2 == 0) { // check even
            count++;
        }
    }

    cout << "Total even numbers = " << count;

    return 0;
}
